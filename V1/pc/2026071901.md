<template>

      <span class="fly-mid"></span>
      <a :class="{ 'layui-this': status === '' && tag === '精华' }" @click.prevent="search(2)"
        >精华</a
      >
      <span class="fly-filter-right layui-hide-xs">
        <a :class="{ 'layui-this': sort === 'created' }" @click.prevent="search(3)">按最新</a>
        <span class="fly-mid"></span>
        <a :class="{ 'layui-this': sort === 'answer' }" @click.prevent="search(4)">按热议</a>
      </span>

  </div>

<script lang="ts" setup>
import { reactive, toRefs } from 'vue'

  status: '',
  tag: '',
  sort: 'created',
})
const { status, tag, sort } = toRefs(query)
const search = (val?: number) => {
  switch (val) {
    case 0:
      query.status = '0'
      query.tag = ''
      break
    case 1:
      query.status = '1'
      query.tag = ''
      break
    case 2:
      query.status = ''
      query.tag = '精华'
      break
    case 3:
      query.sort = 'created'
      break
    case 4:
      query.sort = 'answer'
      break

    default:
      query.status = ''
      query.tag = ''
  }
}
</script>

- 这里的逻辑很简单：综合的status和tag是空也就是默认，未结status是0，已结status是1，精华tag是精华，按最新sort是created，按热议sort是answer
- 通过switch事件，改变状态数据，而`layui-this`又根据这些状态数据而改变，从而实现`layui-this`的交互

<script lang="ts" setup>
import { computed } from 'vue'
import _ from 'lodash'
interface Props {
  list: Array<any> //文章的信息
}
//3.5写法
const { list } = defineProps<Props>()
//计算属性
const items = computed(() => {
  _.map(list, (item) => {
    switch (item.catalog) {
      case 'ask':
        item.catalog = '提问'
        break
      case 'share':
        item.catalog = '分享'
        break
      case 'logs':
        item.catalog = '动态'
        break
      case 'notice':
        item.catalog = '公告'
        break
      case 'advise':
        item.catalog = '建议'
        break
      case 'discuss':
        item.catalog = '交流'
        break
    }
  })
  return list
})

const emit = defineEmits(['next-page'])
const more = () => {
  emit('next-page')
}
</script>

现在有一个问题需要解决，就是时间的格式化，比如显示几分钟前，几小时前，几天前，超过七天的显示几月几号

## 发帖时间格式化

安装dayjs

```shell
npm install dayjs
```

创建：src\utils\formatDate.ts

```ts
import dayjs from 'dayjs'
import 'dayjs/locale/zh-cn'
import relativeTime from 'dayjs/plugin/relativeTime'

dayjs.extend(relativeTime)

export const formatDate = (date: string) => {
  // 超过7天，显示日期
  if (dayjs(date).isBefore(dayjs().subtract(7, 'day'))) {
    return dayjs(date).format('YYYY-MM-DD')
  } else {
    // 1小时前、xx小时前、X天前
    return dayjs(date).locale('zh-cn').from(dayjs())
  }
}

```
