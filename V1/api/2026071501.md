```js
import config from './config/index';
import ErrorHandle from './common/ErrorHandle';

//定义公开的路径
const jwt = JWT({ secret: config.JWT_SECRET }).unless({
  path: [/^\/public/, /^\/login/],
});

// 中间件
// compose集合中间件
const middleware = compose([
  koaBody(),
  cors(),
  json(),
  helmet(),
  ErrorHandle,
  jwt,
  statics(path.join(__dirname, '../public')),
]);
```

src\common\ErrorHandle.js

```js
// 对JWT的错误处理
export default (ctx, next) => {
  return next().catch((err) => {
    if (err.status === 401) {
      ctx.status = 401;
      ctx.body = {
        code: 401,
        msg: 'Protected resource, use Authorization header to get access\n',
      };
    } else {
      // Koa错误
      ctx.status = err.status || 500;
      ctx.body = {
        code: 500,
        msg: err.message,
      };
      // 开发模式调试
      console.log(err.stack);
    }
  });
};

```

- 定义JWT_SECRET，config的index引入
- 定义公开的路径，也就是不需要鉴权的路由，
- 自定义ErrorHandle，对jwt和koa错误统一处理

## 优化登录接口逻辑

src\api\loginController.js

```js
  //登录
  async login(ctx) {
    // 接受用户数据
    const { body } = ctx.request;
    const sid = body.sid; // 图片验证码K值
    const code = body.code; // 图片验证码
    // 验证图片验证码的时效性、正确性
    let result = await cheackCode(sid, code);

    //验证码通过
    if (result) {
      //验证账户名或者密码是否正确
      let chekuserPasword = '';
      const user = await User.findOne({ username: body.username });
      if (user.password === body.password) {
        chekuserPasword = true;
      }
      //查询没有该账号
      if (!user) {
        ctx.body = {
          code: 404,
          msg: '请检查账号！',
        };
        return;
      }
      //验证用户密码
      if (chekuserPasword) {
        // 生成Tonken,有效期1d=>1天
        const token = jsonwebtoken.sign(
          { _id: '_xiaofeng' },
          config.JWT_SECRET,
          {
            expiresIn: '1d',
          },
        );
        ctx.body = {
          code: 200,
          token: token,
          msg: '登录成功',
        };
      } else {
        //用户名密码错误
        ctx.body = {
          code: 404,
          msg: '用户名或者密码错误',
        };
      }
    } else {
      //验证码不通过
      ctx.body = {
        code: 401,
        msg: '图片验证码错误，请检查',
      };
    }
  }
```

- 最开始验证验证码是否正确，否的话返回图片验证码错误，请检查
- 验证码通过，查询数据库有没有该用户，没有该用户，返回404，请检查用户名
- 查询有该用户，对比密码，密码错误，返回404，用户名或者密码错误
- 用户名密码都正确，生成token，并且返回token
